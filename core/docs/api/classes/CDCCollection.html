Class CDCCollection<T>

Wraps a Collection with automatic CDC (Change Data Capture) event emission.

CDCCollection is a transparent wrapper around any collection that automatically emits events for all mutations. This enables:

  • Change Tracking: Every insert, update, and delete is captured
  • Previous State: Optional tracking of document state before changes
  • PITR Support: SQLite bookmarks enable point-in-time recovery
  • Audit Trail: Complete history of all collection changes

The wrapper maintains the same interface as the underlying collection, so it can be used as a drop-in replacement.

Example: Basic Usage

import { EventEmitter, CDCCollection } from '@dotdo/events'

interface User {
  name: string
  email: string
  active: boolean
}

export class UserDO extends DurableObject {
  events = new EventEmitter(this.ctx, this.env, { cdc: true, trackPrevious: true })
  users = new CDCCollection<User>(
    this.collection<User>('users'),
    this.events,
    'users'
  )

  async createUser(user: User) {
    // Emits: { type: 'collection.insert', collection: 'users', docId: user.id, doc: user }
    this.users.put(user.id, user)
  }

  async updateUser(id: string, updates: Partial<User>) {
    const existing = this.users.get(id)
    if (existing) {
      // Emits: { type: 'collection.update', collection: 'users', docId: id, doc: {...}, prev: existing }
      this.users.put(id, { ...existing, ...updates })
    }
  }

  async deleteUser(id: string) {
    // Emits: { type: 'collection.delete', collection: 'users', docId: id, prev: existingDoc }
    this.users.delete(id)
  }
}

Example: Bulk Operations

// Insert multiple documents (emits event for each)
users.putMany([
  { id: 'user-1', doc: { name: 'Alice' } },
  { id: 'user-2', doc: { name: 'Bob' } }
])

// Delete multiple documents (emits event for each)
users.deleteMany(['user-1', 'user-2'])
